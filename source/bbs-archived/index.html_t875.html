Public Function Eggache(ByVal n As Long) As Long
    On Error Resume Next
    Dim lngSum As Long, i As Long
    lngSum = 0
Begin:
    i = i + 1
    On i - n GoTo Done
    lngSum = lngSum + i
    GoTo Begin
Done:
    Eggache = lngSum
End Function
