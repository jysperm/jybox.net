今天写了一个比较简单的游戏引擎wrapper

只是实现了init。如果想做函数方便设计程序的话可以参考。


Sub wTVInitWindow(Engine As TVEngine, wHWnd As Long, wTVEnableTL As Boolean, Optional IsDebug As Boolean = False, Optional IsWriteToConsole As Boolean = False, Optional IsWriteToStdOut As Boolean = False, Optional IsEnableMiniMap As Boolean = False, Optional IsDisplayFPS As Boolean = False)
    Set Engine = New TVEngine
    Engine.Init3DWindowed wHWnd, wTVEnableTL
    If IsDebug = True Then
        Engine.SetDebugMode True, IsWriteToConsole, IsWriteToStdOut, IsEnableMiniMap
        Engine.AddToLog "Wrapper:DebugModeStarted"
    End If
    If IsDisplayFPS = True Then
        Engine.DisplayFPS True
    Else
        Engine.DisplayFPS False
    End If
End Sub
