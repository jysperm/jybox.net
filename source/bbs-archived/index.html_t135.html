PHP的函数重载(伪重载)

字数有限，只贴代码，详见原文。
  1. <?php
  3. /** 零个参数 */
  4. function hello0()
  5. {
  6.     echo 'hello0<br />';
  7. }
  9. /** 一个参数 */
  10. function hello1($param)
  11. {
  12.     echo 'hello1'.$param.'<br />';
  13. }
  15. /** 两个参数 */
  16. function hello2($pa1, $pa2)
  17. {
  18.     echo 'hello2'.$pa1.$pa2.'<br />';
  19. }
  21. /** 三个参数 */
  22. function hello3($p1, $p2, $p3)
  23. {
  24.     echo "hello3{$p1}{$p2}{$p3}<br />";
  25. }
  27. /** 不解释 */
  28. function hello($_=NULL)        // 个人以为$_=NULL就相当于表示参数不定，不赞同的无视它吧
  29. {
  30.     $num = func_num_args();
  31.   
  32.     /**
  33.      * __FUNCTION__取得当前函数名，即'hello'，$num是当前的参数个数
  34.      * __FUNCTION__.$num 即实际调用的函数
  35.      * 关于call_user_func_array大家可以去网上查阅资料，在此不再赘述
  36.      *
  37.      */
  38.     call_user_func_array(__FUNCTION__.$num, func_get_args());
  39. }
  41. // 测试一下---------------------
  42. hello();
  43. hello('param1');
  44. hello('p1', '|', 'p3');
  45. hello("two", '2nd');
  47. ?>

  1. <?php
  3. /**
  4. * 参考：
  5. * 整数:integer, 浮点数:double, 布尔:boolean, 字符串:string, 资源:resourrce
  6. * 数组:array, 对象:object, 未知类型:unknown type;
  7. */
  9. /** 木有参数 */
  10. function hello_void()
  11. {
  12.     echo 'hello without param<br />';
  13. }
  15. /** 一个整数参数 */
  16. function hello_integer($n)
  17. {
  18.     echo $n.'+1='.($n+1).'<br />';
  19. }
  21. /** 两个整数参数 */
  22. function hello_integer_integer($num0, $num1)
  23. {
  24.     echo ($num0+$num1).'<br />';
  25. }
  27. /** 两个字符串参数 */
  28. function hello_string_string($str0, $str1)
  29. {
  30.     echo "{$str0}{$str1}<br />";
  31. }
  33. /** 两个数组参数，一个布尔 */
  34. function hello_array_array_boolean($arr0, $arr1, $b)
  35. {
  36.     if($b)
  37.         print_r($arr0);
  38.     print_r($arr1);
  39.     echo '<br />';
  40. }
  42. /** 不解释。 */
  43. function hello($_=NULL)
  44. {
  45.     // 此函数根据参数的个数和类型动态的选择执行哪个函数
  46.     $__ = '';    // 这个字符串用于添加到函数名的末尾组成真正调用的函数名
  47.     if(!func_num_args())    // 如果没有参数
  48.         $__ = '_void';        // 置$__为_void，则__FUNCTION__.$__就是'hello_void'
  49.     else    // 如果有参数
  50.         foreach(func_get_args() as $arg)    // func_get_args()返回当前参数的一个数组，按顺序遍历之
  51.             $__ .= '_'.gettype($arg);        // 根据当前参数的类型在$__后加上 _<参数类型>
  52.     // 调用__FUNCTION__.$__，以func_get_args()为参数
  53.     // 最后 $__ 实际上就是 _<参数一类型>_<参数二类型>.._<参数n类型>，
  54.     // __FUNCTION__当前函数名即hello, __FUNCTION__.$__就是实际调用的函数
  55.     call_user_func_array( __FUNCTION__.$__ , func_get_args() );
  56. }
  58. hello();
  59. hello(1);
  60. hello(1, 2);
  61. hello('str0', 'str1');
  62. hello(array(0, 1), array(2, 3), false);
  64. ?>
